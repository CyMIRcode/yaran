{{- $imageParam := .Params.image -}} {{/* 1. 自动随机逻辑：如果文章没设 image，自动从 assets/thumbnails 抓取 */}} {{- if not $imageParam -}} {{- if (fileExists "assets/thumbnails") -}} {{- $files := readDir "assets/thumbnails" -}} {{- if $files -}} {{- $randomFile := index (shuffle $files) 0 -}} {{/* 注意：这里生成的路径必须包含 assets 才能被 resources.Get 识别 */}} {{- $imageParam = printf "assets/thumbnails/%s" $randomFile.Name -}} {{- end -}} {{- end -}} {{- end -}} {{/* 2. 渲染逻辑 */}} {{- if $imageParam -}} {{/* 移除可能误输入的首斜杠，确保 resources.Get 正常工作 */}} {{- $path := strings.TrimPrefix "/" $imageParam -}} {{- $res := resources.Get $path -}} {{- if $res -}} {{/* 只有资源存在，才能使用 .Fill 进行缩略图处理 */}} {{- $thumb := $res.Fill "120x120" -}} {{- else -}} {{/* 兜底：如果找不到资源（比如路径写错了），直接作为普通 URL 渲染 */}} {{- end -}} {{- end -}}
